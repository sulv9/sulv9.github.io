int removeDuplicates(int* nums, int numsSize){
    if (numsSize <= 0) return 0;
    int front = 0;
    for (int end = 1; end < numsSize; end++) {
        if (nums[end] != nums[front]) {
            front++;
            nums[front] = nums[end];
        }
    }
    return front + 1;
}
